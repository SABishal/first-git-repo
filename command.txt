git init

#2. every time to add something
git add .

#3. every time wants to commit
git commit -m "first commit"

#4. good practice
git branch -M main

#5. where to push the codes
git remote add origin https://github.com/SABishal/first-git-repo.git

#6. need to use for pushing code every time
git push -u origin main
